## Yet another cses problem

This time, we solve distinct numbers, found here: https://cses.fi/problemset/task/1621/

```av
void* alloca(int32 bytes);
int32 read(int32 fileno, void* mem, int32 bytes);
int32 write(int32 fileno, void* mem, int32 bytes);

int8 read_int8(void* buf, int32* loc, int32 size) {
  int32 l = *loc;
  if (size == l) {
    return 0;
  }
  int8* ptr = buf + l;
  int8 c = *ptr;
  *loc = l + 1;
  return c;
}

void write_int8(int8 c, void* buf, int32* loc) {
  int32 l = *loc;
  int8* ptr = buf + l;
  *ptr = c;
  *loc = l + 1;
}

int64 read_int64(void* buf, int32* loc, int32 size) {
  int8 c = read_int8(buf, loc, size);
  while (c <= 32) {
    c = read_int8(buf, loc, size);
  }
  int64 ans = 0;
  while (48 <= c & c <= 57) {
    ans = (ans << 3) + (ans << 1) + c - 48;
    c = read_int8(buf, loc, size);
  }
  return ans;
}

void write_int64(int64 x, void* buf, int32* loc) {
  if (x == 0) {
    write_int8(48, buf, loc);
    return;
  }
  int8* s = alloca(20);
  int32 i = 0;
  while (x) {
    int8* ptr = s + i;
    *ptr = (x % 10) + 48;
    x = x / 10;
    i = i + 1;
  }
  while (i) {
    i = i - 1;
    int8* ptr = s + i;
    int8 c = *ptr;
    write_int8(c, buf, loc);
  }
}

int32 arr(void* a, int32 i) {
  int32* ptr = a + 4 * i;
  int32 x = *ptr;
  return x;
}

void sort(void* a, int32 l, int32 r) {
  if (l >= r) {
    return;
  }
  int32 m = (l + r) / 2;
  sort(a, l, m);
  int32 m_plus_one = m + 1;
  sort(a, m_plus_one, r);

  int32 sz = (r - l + 1) << 2;
  void* t = alloca(sz);
  int32 ptr1 = l;
  int32 ptr2 = m + 1;
  int32 i = 0;
  while (ptr1 <= m & ptr2 <= r) {
    int32* ptr = t + 4 * i;
    int32 v1 = arr(a, ptr1);
    int32 v2 = arr(a, ptr2);
    if (v1 < v2) {
      *ptr = arr(a, ptr1);
      ptr1 = ptr1 + 1;
    } else {
      *ptr = arr(a, ptr2);
      ptr2 = ptr2 + 1;
    }
    i = i + 1;
  }
  while (ptr1 <= m) {
    int32* ptr = t + 4 * i;
    *ptr = arr(a, ptr1);
    ptr1 = ptr1 + 1;
    i = i + 1;
  }
  while (ptr2 <= r) {
    int32* ptr = t + 4 * i;
    *ptr = arr(a, ptr2);
    ptr2 = ptr2 + 1;
    i = i + 1;
  }

  for (int32 ii = l; ii <= r; ii = ii + 1) {
    int32* ptr = a + 4 * ii;
    int32 loc = ii - l;
    *ptr = arr(t, loc);
  }
}

int32 main() {
  int32 buf_size = 5000000;
  void* rbuf = alloca(buf_size);
  void* wbuf = alloca(buf_size);
  int32 rloc_ = 0;
  int32 wloc_ = 0;
  int32* rloc = &rloc_;
  int32* wloc = &wloc_;
  int32 rsize = read(0, rbuf, buf_size);

  int32 n = read_int64(rbuf, rloc, rsize);
  int32 size = n << 2;
  void* a = alloca(size);
  for (int32 i = 0; i < n; i = i + 1) {
    int32 x = read_int64(rbuf, rloc, rsize);
    int32* ptr = a + 4 * i;
    *ptr = x;
  }

  int32 n_minus_one = n - 1;
  sort(a, 0, n_minus_one);

  int32 ans = 1;
  for (int32 i = 1; i < n; i = i + 1) {
    int32 v1 = arr(a, i);
    int32 i_minus_one = i - 1;
    int32 v0 = arr(a, i_minus_one);
    if (v1 == v0) {
      ans = ans;
    } else {
      ans = ans + 1;
    }
  }
  write_int64(ans, wbuf, wloc);
  write_int8(10, wbuf, wloc);

  write(1, wbuf, wloc_);
  return 0;
}
```
