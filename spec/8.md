## Missing number

This `.av` code solves missing number on cses.fi:

```av
int32 read(int32 fileno, void* buf, int32 bytes);
int32 write(int32 fileno, void* buf, int32 bytes);
void* alloca(int32 bytes);

int8 getchar(void* buf, int32* loc) {
  int32 actual_loc = *loc;
  int8* ptr = buf + actual_loc;
  int8 x = *ptr;
  *loc = actual_loc + 1;
  return x;
}

void putchar(int8 c) {
  int8* ptr = &c;
  write(1, ptr, 1);
}

int32 is_digit(int8 x) {
  return x >= 48 & x <= 57;
}

int64 read_int(void* buf, int32* loc) {
  int64 x = 0;
  int8 c = getchar(buf, loc);
  while (is_digit(c)) {
    x = 10 * x + c - 48;
    c = getchar(buf, loc);
  }
  return x;
}

void print_int(int64 x) {
  if (x == 0) {
    putchar(48);
    return;
  }

  void* mem = alloca(30);
  int32 i = 0;

  while (x) {
    int8* ptr = mem + i;
    int64 d = x % 10;
    x = x / 10;
    *ptr = d + 48;
    i = i + 1;
  }

  while (i) {
    int8* ptr = mem + i - 1;
    int8 c = *ptr;
    putchar(c);
    i = i - 1;
  }
}

int32 main() {
  void* buf = alloca(3600000);
  int32 loc = 0;
  int32* locptr = &loc;
  read(0, buf, 3600000);

  int64 n = read_int(buf, locptr);
  int64 sum = 0;
  int32 i = 0;
  while (i < n - 1) {
    sum = sum + read_int(buf, locptr);
    i = i + 1;
  }
  int64 n_plus_one = n + 1;
  int64 prod1 = n * n_plus_one;
  int64 missing = prod1 / 2 - sum;
  print_int(missing);
  putchar(10);
  return 0;
}
```
