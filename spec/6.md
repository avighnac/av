An interesting use of inheritance yet again:
```cpp
struct Binary : Node {
  Node *Lhs;
  Node *Rhs;
  Binary(const NodeType &type) : Node(type) {}
  ~Binary() {
    delete Lhs;
    delete Rhs;
  }
};

struct Equal : Binary {
  Equal() : Binary(equal) {}
};

struct Less : Binary {
  Less() : Binary(less) {}
};
```
